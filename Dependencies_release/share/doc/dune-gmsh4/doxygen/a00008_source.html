13 inline void swapBytes (char *array, int size)
14 {
15  char *tmp = new char[size];
16  memcpy(tmp, array, size);

295 template <class G, class C, class S>
296 void Gmsh4Reader<G,C,S>::readSerialFileFromStream (std::ifstream& input, bool fillCreator)
297 {
298  clear();
299 
300  // MeshFormat section
301  double version = 0.0;
302  int file_type = 0;
303  int data_size = 0;
304 
305  Sections section = Sections::NO_SECTION;
306  for (std::string line; std::getline(input, line); ) {
307  Gmsh4::ltrim(line);
308 
309  // detect current section
310  for (auto const& s : sections_) {
311  if (isSection(line, s.first, section)) {
312  section = s.second;
313  break;
314  } else if (isSection(line, "End" + s.first, section, s.second)) {
315  section = Sections::NO_SECTION;
316  break;
317  }
318  }
319 
320  switch (section) {
321  case Sections::MESH_FORMAT: {
322  readMeshFormat(input, version, file_type, data_size);
323  GMSH4_ASSERT_MSG(version >= 4.0 && version < 5.0,
324  "Can only read gmsh files versions >= 4.0 and < 5.0");
325  GMSH4_ASSERT_MSG(file_type == 0 || file_type == 1,
326  "Invalid file-type: 0 for ASCII mode, 1 for binary mode");
327  GMSH4_ASSERT_MSG(data_size >= 4 && data_size <= 16,
328  "Invalid data-size range: should be in {4, 16}");
329  GMSH4_ASSERT_MSG(file_type != 1 || data_size == sizeof(size_type),
330  "Invalid data-size: must be sizeof(size_t)");
331  break;
332  }
333  case Sections::PHYSICAL_NAMES:
334  readPhysicalNames(input); break;
335  case Sections::ENTITIES:
336  if(file_type == 0) readEntitiesAscii(input);
337  else readEntitiesBinary(input);
338  break;
339  case Sections::PARTITIONED_ENTITIES:
340  if(file_type == 0) readPartitionedEntitiesAscii(input);
341  else readPartitionedEntitiesBinary(input);
342  break;
343  case Sections::NODES:
344  if(file_type == 0) readNodesAscii(input);
345  else readNodesBinary(input);
346  break;
347  case Sections::ELEMENTS:
348  if(file_type == 0) readElementsAscii(input);
349  else readElementsBinary(input);
350  break;
351  case Sections::PERIODIC:
352  readPeriodic(input); break;
353  case Sections::GHOST_ELEMENTS:
354  readGhostElements(input); break;
355  case Sections::PARAMETRIZATION:
356  readParametrization(input); break;
357  case Sections::NODE_DATA:
358  readNodeData(input); break;
359  case Sections::ELEMENT_DATA:
360  readElementData(input); break;
361  case Sections::ELEMENT_NODE_DATA:
362  readElementNodeData(input); break;
363  case Sections::INTERPOLATION_SCHEME:
364  readInterpolationScheme(input); break;
365  default:
366  // do nothing
367  break;
368  }
369  }
370 
371  if (fillCreator)
372  fillGridCreator();
373 }

153  void setup1d(std::vector<unsigned int>& subEntity) {
154  if (sizes_[0] == 1) {
155  subEntity[0] = 0;
156  return;
157  }
158 
159  /* edge and vertex numbering
160  0----0----1
161  */
162  unsigned lastIndex = 0;
163  subEntity[lastIndex++] = 0; // corner 0
164  for (unsigned i = 0; i < sizes_[0] - 2; ++i)
165  subEntity[lastIndex++] = 0; // inner dofs of element (0)
166 
167  subEntity[lastIndex++] = 1; // corner 1
168 
169  assert(size() == lastIndex);
170  }

172  void setup2d(std::vector<unsigned int>& subEntity) {
173  unsigned lastIndex = 0;
174 
175  // LocalKey: entity number , entity codim, dof indices within each entity
176  /* edge and vertex numbering
177  2----3----3
178  | |
179  | |
180  0 1
181  | |
182  | |
183  0----2----1
184  */
185 
186  // lower edge (2)
187  subEntity[lastIndex++] = 0; // corner 0
188  for (unsigned i = 0; i < sizes_[0] - 2; ++i)
189  subEntity[lastIndex++] = 2; // inner dofs of lower edge (2)
190 
191  subEntity[lastIndex++] = 1; // corner 1
192 
193  // iterate from bottom to top over inner edge dofs
194  for (unsigned e = 0; e < sizes_[1] - 2; ++e) {
195  subEntity[lastIndex++] = 0; // left edge (0)
196  for (unsigned i = 0; i < sizes_[0] - 2; ++i)
197  subEntity[lastIndex++] = 0; // face dofs
198  subEntity[lastIndex++] = 1; // right edge (1)
199  }
200 
201  // upper edge (3)
202  subEntity[lastIndex++] = 2; // corner 2
203  for (unsigned i = 0; i < sizes_[0] - 2; ++i)
204  subEntity[lastIndex++] = 3; // inner dofs of upper edge (3)
205 
206  subEntity[lastIndex++] = 3; // corner 3
207 
208  assert(size() == lastIndex);
209  }

212  void init(const std::array<unsigned, dim>& sizes) {
213  sizes_ = sizes;
214 
215  li_.resize(size());
216 
217  // Set up array of codimension-per-dof-number
218  std::vector<unsigned int> codim(li_.size());
219 
220  for (std::size_t i = 0; i < codim.size(); i++) {
221  codim[i] = 0;
222  // Codimension gets increased by 1 for each coordinate direction
223  // where dof is on boundary
224  std::array<unsigned int, dim> mIdx = multiindex(i);
225  for (int j = 0; j < dim; j++)
226  if (mIdx[j] == 0 or mIdx[j] == sizes[j] - 1) codim[i]++;
227  }
228 
229  // Set up index vector (the index of the dof in the set of dofs of a given subentity)
230  // Algorithm: the 'index' has the same ordering as the dof number 'i'.
231  // To make it consecutive we interpret 'i' in the (k+1)-adic system, omit all digits
232  // that correspond to axes where the dof is on the element boundary, and transform the
233  // rest to the (k-1)-adic system.
234  std::vector<unsigned int> index(size());
235 
236  for (std::size_t i = 0; i < index.size(); i++) {
237  index[i] = 0;
238 
239  std::array<unsigned int, dim> mIdx = multiindex(i);
240 
241  for (int j = dim - 1; j >= 0; j--)
242  if (mIdx[j] > 0 and mIdx[j] < sizes[j] - 1) index[i] = (sizes[j] - 1) * index[i] + (mIdx[j] - 1);
243  }
244 
245  // Set up entity and dof numbers for each (supported) dimension separately
246  std::vector<unsigned int> subEntity(li_.size());
247 
248  if (!subEntity.empty()) {
249  if (dim == 1) {
250  setup1d(subEntity);
251 
252  } else if (dim == 2 and sizes_[0] > 1 and sizes_[1] > 1) {
253  setup2d(subEntity);
254  }
255  }
256 
257  for (size_t i = 0; i < li_.size(); i++)
258  li_[i] = LocalKey(subEntity[i], codim[i], index[i]);
259  }
260 
262  [[nodiscard]] std::size_t size() const { return std::accumulate(sizes_.begin(), sizes_.end(), 1, std::multiplies<>()); }
263 
265  [[nodiscard]] const LocalKey& localKey(std::size_t i) const { return li_[i]; }

501  // Local degrees of freedom are arranged in a lattice.

503  std::array<unsigned int, dim> localSizes;
504  for (int i = 0; i < dim; i++)

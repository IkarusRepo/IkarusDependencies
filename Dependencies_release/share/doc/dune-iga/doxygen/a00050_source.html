79  [[nodiscard]] GlobalCoordinate outerNormal(const LocalCoordinate& xi) const {

82  const auto insideJacobianTransposed = inside().geometry().jacobianTransposed(xiInElementCoords);
83  return (innerLocalIndex_ == 0) ? -insideJacobianTransposed[0] : insideJacobianTransposed[0];
84  } else if constexpr (mydimension == 1) { // edges
85  if constexpr (dimworld == 2) { // edges in R2
86  const auto innerJacobianTransposed = this->geometry().jacobianTransposed(xi)[0];
87  switch (innerLocalIndex_) {
88  case 0:
89  case 3: return GlobalCoordinate({-innerJacobianTransposed[1], innerJacobianTransposed[0]});
90  case 1:

92  default: __builtin_unreachable();
93  }

95  {

98  auto normal = cross(insideJacobianTransposed[0], insideJacobianTransposed[1]);
99  switch (innerLocalIndex_) {

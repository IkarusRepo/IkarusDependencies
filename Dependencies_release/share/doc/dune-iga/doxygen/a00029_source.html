104  [[nodiscard]] LocalCoordinate local(const GlobalCoordinate& global) const {
105  const ctype tolerance = ctype(16) * std::numeric_limits<ctype>::epsilon();
106  LocalCoordinate x = LocalCoordinate(0.5);
107  LocalCoordinate dx{};
108  do { // from multilinearGeometry
109  const GlobalCoordinate dglobal = (*this).global(x) - global;
110  MatrixHelper::template xTRightInvA<mydimension, coorddimension>(jacobianTransposed(x), dglobal, dx);
111  const bool invertible = MatrixHelper::template xTRightInvA<mydimension, coorddimension>(jacobianTransposed(x), dglobal, dx);
112 
113  if (!invertible) return LocalCoordinate(std::numeric_limits<ctype>::max());
114  x -= dx;
115  } while (dx.two_norm2() > tolerance);
116  return x;

188  auto secondDerivativeOfPosition(const LocalCoordinate& local) const {
189  FieldMatrix<ctype, coorddimension, mydimension*(mydimension + 1) / 2> result;
190  std::array<unsigned int, mydimension> subDirs;
191  for (int subI = 0, i = 0; i < griddim; ++i) {
192  if (fixedOrVaryingDirections_[i] == Dune::IGA::Impl::FixedOrFree::fixed) continue;
193  subDirs[subI++] = i;
194  }
195  const auto localInSpan = transformLocalToSpan(local);
196  const auto basisFunctionDerivatives = nurbs_.basisFunctionDerivatives(localInSpan, 2);
197  for (int dir = 0; dir < mydimension; ++dir) {
198  std::array<unsigned int, griddim> ithVec{};
199  ithVec[subDirs[dir]] = 2; // second derivative in dir direction
200  result[dir] = dot(basisFunctionDerivatives.get(ithVec), cpCoordinateNet_);
201  result[dir] *= Dune::power(scaling_[subDirs[dir]], 2); // transform back to 0..1
202  }
203  std::array<int, mydimension> mixeDerivs;
204  std::ranges::fill_n(mixeDerivs.begin(), 2, 1); // first mixed derivatives
205  int mixedDireCounter = mydimension;
206  do {
207  result[mixedDireCounter++] = dot(basisFunctionDerivatives.get(mixeDerivs), cpCoordinateNet_);
208  for (int dir = 0; dir < mixeDerivs.size(); ++dir) {
209  if (mixeDerivs[dir] == 0) continue;
210  result[mixedDireCounter - 1] *= scaling_[subDirs[dir]];
211  }
212  } while (std::ranges::next_permutation(mixeDerivs, std::greater()).found);
213 
214  return result;
215  }

159  static auto basisFunctionDerivatives(std::array<ScalarType, dim> u, const std::array<std::vector<ScalarType>, dim>& knots,
160  const std::array<int, dim>& degree, const MultiDimensionNet<dim, double>& weights,
161  const int derivativeOrder, const bool triangleDerivatives = false,
162  std::optional<std::array<int, dim>> spIndex = std::nullopt) {
163  std::array<typename NurbsGridLinearAlgebraTraits::DynamicMatrixType, dim> bSplineDerivatives;
164  for (int i = 0; i < dim; ++i)
165  bSplineDerivatives[i] = BsplineBasis1D<ScalarType>::basisFunctionDerivatives(
166  u[i], knots[i], degree[i], derivativeOrder, (spIndex ? std::optional<int>(spIndex.value()[i]) : std::nullopt));
167 
168  std::array<std::vector<ScalarType>, dim> dimArrayOfVectors;
169  FieldVector<int, dim> dimSize(derivativeOrder + 1);
170  MultiDimensionNet<dim, MultiDimensionNet<dim, ScalarType>> netOfDerivativeNets(dimSize);
171 
172  for (int j = 0; auto& derivNet : netOfDerivativeNets.directGetAll()) {
173  auto multiIndex = netOfDerivativeNets.directToMultiIndex(j++);
174  for (int i = 0; i < multiIndex.size(); ++i)
175  dimArrayOfVectors[i] = bSplineDerivatives[i][multiIndex[i]].container();
176  derivNet = MultiDimensionNet<dim, ScalarType>(dimArrayOfVectors);
177  }
178 
179  MultiDimensionNet<dim, MultiDimensionNet<dim, ScalarType>> R = netOfDerivativeNets;
180  MultiDimensionNet<dim, ScalarType> netsOfWeightfunctions(dimSize);
181  auto subNetStart = spIndex ? spIndex.value() : findSpan(degree, u, knots);
182  const auto subNetWeights = netOfSpan(subNetStart, degree, weights);
183 
184  for (int j = 0; j < R.directSize(); ++j) {
185  R.directGet(j) *= subNetWeights;
186  netsOfWeightfunctions.directGet(j) = dot(netOfDerivativeNets.directGet(j), subNetWeights);
187  }
188 
189  for (int j = 0; j < R.directSize(); ++j) {
190  const auto derivOrders = R.template directToMultiIndex<FieldVector<int, dim>>(j);
191  if (triangleDerivatives)
192  if (std::accumulate(derivOrders.begin(), derivOrders.end(), derivativeOrder)) continue;
193  const auto perms = Impl::createPartialSubDerivativPermutations(derivOrders);
194 
195  for (const auto& perm : perms) {
196  MultiDimensionNet<dim, int> kNet(perm + FieldVector<int, dim>(1));
197  auto startMultiIndex = perm;
198  std::ranges::transform(startMultiIndex, startMultiIndex.begin(), [](auto& v) { return (v != 0); });
199 
200  for (int kk = kNet.index(startMultiIndex); kk < kNet.directSize(); ++kk) {
201  const auto multik = kNet.template directToMultiIndex<FieldVector<int, dim>>(kk);
202  R.directGet(j) -= Impl::binom(perm, multik)
203  * (R.get(derivOrders - multik) * netsOfWeightfunctions.get(multik)); // generalized Piegl Tiller (4.20)
204  }
205  }
206  R.directGet(j) /= netsOfWeightfunctions.directGet(0);
207  }
208  return R;
209  }

223  template <std::integral auto dim, std::integral auto dimworld, LinearAlgebra NurbsGridLinearAlgebraTraitsImpl>
224  auto degreeElevate(const NURBSPatchData<dim, dimworld, NurbsGridLinearAlgebraTraitsImpl>& oldData, const int refinementDirection,
225  const int elevationFactor) {
226  using DynamicMatrixType = typename NurbsGridLinearAlgebraTraitsImpl::DynamicMatrixType;
227  using ScalarType = typename NurbsGridLinearAlgebraTraitsImpl::value_type;
228  const int t = elevationFactor;
229  const int p = oldData.degree[refinementDirection];
230  const int m = oldData.controlPoints.size()[refinementDirection] + p;
231  const int ph = p + t;
232  const int ph2 = ph / 2;
233  /* Compute Bezier degree elevation coefficients */
234  DynamicMatrixType bezalfs(p + t + 1, p + 1);
235  bezalfs[0][0] = bezalfs[ph][p] = 1.0;
236  for (int i = 1; i <= ph2; ++i) {
237  const ScalarType inv = 1.0 / Dune::binomial(ph, i);
238  for (int j = std::max(0, i - t); j <= std::min(p, i); ++j)
239  bezalfs[i][j] = inv * Dune::binomial(p, j) * Dune::binomial(t, i - j);
240  }
241  for (int i = ph2 + 1; i <= ph - 1; ++i)
242  for (int j = std::max(0, i - t); j <= std::min(p, i); ++j)
243  bezalfs[i][j] = bezalfs[ph - i][p - j];
244 
245  double ua = oldData.knotSpans[refinementDirection][0];
246  typename NURBSPatchData<dim, dimworld, NurbsGridLinearAlgebraTraitsImpl>::ControlPointNetType newCPv(oldData.controlPoints.size());
247  using ControlPointType = typename NURBSPatchData<dim, dimworld, NurbsGridLinearAlgebraTraitsImpl>::ControlPointType;
248 
249  auto oldCPs = oldData.controlPoints;
250 
251  auto scaleCPWithW = [](const auto& cp) -> ControlPointType { return {.p = cp.w * cp.p, .w = cp.w}; };
252  std::ranges::transform(oldCPs.directGetAll(), oldCPs.directGetAll().begin(), scaleCPWithW);
253 
254  std::vector<ControlPointType> bpts(p + 1), ebpts(ph + 1), nextbpts(p - 1);
255 
256  NURBSPatchData<dim, dimworld, NurbsGridLinearAlgebraTraitsImpl> newData;
257  newData.degree = oldData.degree;
258  newData.degree[refinementDirection] += t;
259  const auto& U = oldData.knotSpans[refinementDirection];
260  std::vector<ScalarType> Uh;
261  for (int j = 0, i = 0; i < U.size(); ++i) // insert knot t times for each unique knot
262  if (j < U.size()) {
263  do {
264  Uh.push_back(U[j]);
265  ++j;
266  } while (j != U.size() && Dune::FloatCmp::eq(U[j - 1], U[j]));
267  Uh.insert(Uh.end(), t, U[j - 1]);
268  if (j == U.size()) break;
269  }
270 
271  for (int i = 0; i < dim; ++i)
272  if (i == refinementDirection)
273  newData.knotSpans[i] = Uh;
274  else
275  newData.knotSpans[i] = oldData.knotSpans[i];
276 
277  auto dimSize = oldData.controlPoints.size();
278  dimSize[refinementDirection] = Uh.size() - ph - 1;
279  newData.controlPoints = MultiDimensionNet<dim, ControlPointType>(dimSize);
280  auto& newCPs = newData.controlPoints;
281 
282  int totalCurvesInPatch = 1;
283  std::array<int, dim - 1> indicesOther;
284  for (int counter = 0, i = 0; i < dim; ++i)
285  if (i != refinementDirection) {
286  totalCurvesInPatch *= dimSize[i];
287  indicesOther[counter++] = dimSize[i];
288  }
289 
290  auto newAndOldCurve = [&refinementDirection, indicesOther, &oldCPs, &newCPs](auto i) {
291  const auto mI = MultiDimensionNet<dim - 1, int>::template directToMultiIndex<std::array<int, dim - 1>>(indicesOther, i);
292  std::array<int, dim> multiIndex{};
293  for (int c = 0, j = 0; j < dim; ++j)
294  if (j != refinementDirection) multiIndex[j] = mI[c++];
295 
296  auto oldCurve = [&oldCPs, &refinementDirection, multiIndex](int i) mutable {
297  multiIndex[refinementDirection] = i;
298  return oldCPs.get(multiIndex);
299  };
300 
301  auto newCurve = [&newCPs, &refinementDirection, multiIndex ](int i) mutable -> auto& {
302  multiIndex[refinementDirection] = i;
303  return newCPs.get(multiIndex);
304  };
305  return std::make_pair(oldCurve, newCurve);
306  };
307 
308  for (auto [oldCurve, newCurve] : std::ranges::iota_view(0, totalCurvesInPatch) | std::views::transform(newAndOldCurve)) {
309  int kind = ph + 1;
310  int r = -1;
311  int a = p;
312  int b = p + 1;
313  int cind = 1;
314 
315  newCurve(0) = oldCurve(0);
316  for (int i = 0; i <= p; ++i)
317  bpts[i] = oldCurve(i);
318 
319  std::vector<ScalarType> alphas(p - 1);
320  while (b < m) {
321  const int bold = b;
322  while (b < m && Dune::FloatCmp::eq(U[b], U[b + 1]))
323  ++b;
324  const int mul = b - bold + 1;
325  const auto ub = U[b];
326  const int oldr = r;
327  r = p - mul;
328  const int lbz = (oldr > 0) ? ((oldr + 2) / 2) : 1; // Insert knot u(b) r times
329  const int rbz = (r > 0) ? (ph - (r + 1) / 2) : ph;
330  if (r > 0) // Insert knot to get Bezier segment
331  {
332  const auto numer = ub - ua;
333  for (int k = p; k > mul; --k)
334  alphas[k - mul - 1] = numer / (U[a + k] - ua);
335  for (int j = 1; j <= r; ++j) {
336  const int s = mul + j;
337  for (int k = p; k >= s; --k)
338  bpts[k] = alphas[k - s] * bpts[k] + (1.0 - alphas[k - s]) * bpts[k - 1];
339  nextbpts[r - j] = bpts[p];
340  }
341  } // End of insert knot
342 
343  for (int i = lbz; i <= ph; ++i) { // Degree elevate Bezier
344  ebpts[i].setZero();
345  for (int j = std::max(0, i - t); j <= std::min(p, i); ++j)
346  ebpts[i] += bezalfs[i][j] * bpts[j];
347  } // End of degree elevating Bezier
348  if (oldr > 1) { // Must remove knot u = U[a] oldr times
349  int first = kind - 2;
350  int last = kind;
351  const ScalarType den = ub - ua;
352  const ScalarType bet = (ub - Uh[kind - 1]) / den;
353  for (int tr = 1; tr < oldr; ++tr) { // Knot removal loop
354  int i = first;
355  int j = last;
356  int kj = j - kind + 1;
357  while (j - i > tr) {
358  if (i < cind) {
359  const ScalarType alf = (ub - Uh[i]) / (ua - Uh[i]);
360  newCurve(i) = alf * newCurve(i) + (1.0 - alf) * newCurve(i - 1);
361  }
362  if (j >= lbz) {
363  const auto factor = (j - tr <= kind - ph + oldr) ? (ub - Uh[j - tr]) / den : bet;
364  ebpts[kj] = factor * ebpts[kj] + (1.0 - factor) * ebpts[kj + 1];
365  }
366  ++i;
367  --j;
368  --kj;
369  }
370  --first;
371  ++last;
372  }
373  } // End of removing knot, u= U[a]
374  if (a != p) kind += ph - oldr;
375  for (int j = lbz; j <= rbz; ++j, ++cind) // Load ctrl pts into Qw
376  newCurve(cind) = ebpts[j];
377  if (b < m) { // Set up for next pass thru loop
378  std::ranges::copy(nextbpts | std::views::take(r), bpts.begin());
379  for (int j = r; j <= p; ++j)
380  bpts[j] = oldCurve(b - p + j);
381  a = b;
382  ++b;
383  ua = ub;
384  }
385  } // End of while-loop (b<m)
386  }
387  std::ranges::transform(newData.controlPoints.directGetAll(), newData.controlPoints.directGetAll().begin(),
388  [](auto& cp) -> ControlPointType { return {.p = cp.p / cp.w, .w = cp.w}; });
389  return newData;
390  }

393  auto knotRefinement(const NURBSPatchData<dim, dimworld, NurbsGridLinearAlgebraTraitsImpl>& oldData, const std::vector<double>& newKnots,
394  const int refinementDirection) {

399  using namespace std::ranges;

401  std::array<int, dim - 1> otherDirections;
402  for (int counter = 0, i = 0; i < dim; ++i) {

412

416  merge(oldKnotVec, newKnots, std::back_inserter(newKnotVec));
417 
418  const auto newKSize = newKnots.size();
419  std::array<int, dim> numberOfCPperDir = oldCPv.size();
420  numberOfCPperDir[refinementDirection] += newKSize;
421 
422  newCPv.resize(numberOfCPperDir);
423 
424  auto scaleCPWithW = [](const auto& cp) -> ControlPoint { return {.p = cp.w * cp.p, .w = cp.w}; };
425 
426  const int p = oldData.degree[refinementDirection];
427  const int a = findSpan(p, newKnots.front(), oldKnotVec);
428  const int b = findSpan(p, newKnots.back(), oldKnotVec);
429 
430  auto newSurfI = newCPv.hyperSurfBegin(otherDirections);
431  auto oldSurfI = oldCPv.hyperSurfBegin(otherDirections);
432 
433  for (auto hSOld = oldSurfI, hSNew = newSurfI; hSOld != oldCPv.hyperSurfBegin(otherDirections) + (a - p + 1); ++hSOld, ++hSNew)
434  std::ranges::transform(*hSOld, (*hSNew).begin(), scaleCPWithW);
435  for (auto hSOld = oldSurfI + b, hSNew = newSurfI + b; hSOld != oldCPv.hyperSurfEnd(otherDirections); ++hSOld, ++hSNew)
436  std::ranges::transform(*hSOld, (*hSNew).begin(), scaleCPWithW);
437 
438  newSurfI = newCPv.hyperSurfEnd(otherDirections) - 1;
439  oldSurfI = oldCPv.hyperSurfEnd(otherDirections) - 1;
440 
441  auto currentNewKnot = newKnotVec.end();
442  auto currentOldKnot = oldKnotVec.end();
443 
444  for (auto const& currentAdditionalKnot : reverse_view(newKnots)) {
445  while (currentAdditionalKnot <= *(currentOldKnot - 1) && std::distance(oldKnotVec.begin(), currentOldKnot) > a + 1) {
446  std::ranges::transform(*oldSurfI, (*newSurfI).begin(), scaleCPWithW);
447  --currentNewKnot, --currentOldKnot;
448  --newSurfI, --oldSurfI;
449  }
450 
451  std::ranges::copy(*(newSurfI + 1), (*newSurfI).begin());
452  ++newSurfI;
453  currentOldKnot -= p;
454  for ([[maybe_unused]] int _ : iota_view{0, p}) {
455  ++newSurfI;
456 
457  auto alpha = *currentNewKnot - currentAdditionalKnot;
458  using std::abs;
459  if (abs(alpha) < 1e-7)
460  std::ranges::copy(*newSurfI, (*(newSurfI - 1)).begin());
461  else {
462  alpha /= (*currentNewKnot - *(currentOldKnot));
463  std::ranges::transform(*newSurfI, *(newSurfI - 1), (*(newSurfI - 1)).begin(),
464  [&alpha](auto& cp, auto& cpL) { return alpha * cpL + cp * (1.0 - alpha); });
465  }
466  ++currentOldKnot, ++currentNewKnot;
467  }
468  currentNewKnot -= p + 1;
469  newSurfI -= p + 2;
470  }
471 
472  std::ranges::transform(newCPv.directGetAll(), newCPv.directGetAll().begin(),
473  [](auto& cp) -> ControlPoint { return {.p = cp.p / cp.w, .w = cp.w}; });
474 
475  return NURBSPatchData<dim, dimworld, NurbsGridLinearAlgebraTraitsImpl>(newKnotsArray, newCPv, oldData.degree);
476  }

492  auto Intersect3DLines(const VectorType basepoint1, const VectorType direction1, const VectorType basepoint2,
493  const VectorType direction2) {
494  using ScalarType = typename VectorType::value_type;
495  const ScalarType tol = 1e-8;
496  const VectorType basePointDifference = basepoint1 - basepoint2;
497  using Dune::dot;
498  const ScalarType dir1squaredLength = dot(direction1, direction1);
499  const ScalarType angle = dot(direction1, direction2);
500  const ScalarType dir2squaredLength = dot(direction2, direction2);
501  const ScalarType dir1BasePdiff = dot(direction1, basePointDifference);
502  const ScalarType dir2BasePdiff = dot(direction2, basePointDifference);
503  const ScalarType D = dir1squaredLength * dir2squaredLength - angle * angle;
504 
505  if (std::abs(D) < tol) throw std::logic_error("The two lines are almost parallel.");
506 
507  const ScalarType parameter1 = (angle * dir2BasePdiff - dir2squaredLength * dir1BasePdiff) / D;
508  const ScalarType parameter2 = (dir1squaredLength * dir2BasePdiff - angle * dir1BasePdiff) / D;
509 
510  VectorType c1 = basepoint1 + parameter1 * direction1;
511  assert(dot(c1 - (basepoint2 + parameter2 * direction2), c1 - (basepoint2 + parameter2 * direction2)) < tol
512  && "Both calculated points do not coincide");
513  return c1;
514  }

598  } else if (revolutionAngle <= 270.0) {

600  U[5] = U[6] = 2.0 / 3.0;

611  generatrix.controlPoints.size()[0]);
612  using std::cos;
613  using std::sin;
614  const ScalarType wm = cos(dtheta / 2.0);
615  std::vector<ScalarType> cosines(narcs);
616  std::vector<ScalarType> sines(narcs);

619  angle += dtheta;

626  GlobalCoordinateType X = genCP.directGet(j).p - Om;

630  surfaceCP(0, j) = PO = genCP.directGet(j);

633  const GlobalCoordinateType P2 = Om + r * cosines[i] * X + r * sines[i] * Y;
634  surfaceCP(index + 2, j) = {.p = P2, .w = genCP.directGet(j).w};

637  surfaceCP(index + 1, j).p = Impl::Intersect3DLines(PO.p, TO, P2, T2);
638  surfaceCP(index + 1, j).w = wm * genCP.directGet(j).w;
639  index += 2;
640  if (i < narcs - 1) {
641  PO.p = P2;
642  TO = T2;
643  }
644  }
645  }

109  int index = ocdim1Id;
110 
111  if constexpr (dim == 1) { // early exit curves only have a left or right boundary point
112  if (ocdim1Id == 0) return 0;
113  if (ocdim1Id == uniqueSpanSize_[0]) return 1;
114  } else {
115  const auto codimSizes = this->sizeOfCodim1PerDirection<1>();
116 
117  int surfFixedDir = 0;
118  int edgesPerDir = codimSizes[0];
119  for (int j = 0; j < dim; ++j)
120  if (ocdim1Id < edgesPerDir) { // find out to which edge direction this edge index belongs
121  surfFixedDir = j;
122  break;
123  } else
124  edgesPerDir += codimSizes.at(j + 1);
125 
126  // move index to correct subset, e.g. edges in y-dir or surfaces with fixed z-dir
127  index -= std::accumulate(codimSizes.begin(), codimSizes.begin() + surfFixedDir, 0);
128 
129  if constexpr (dim == 2) {
130  if (surfFixedDir == 1) {
131  const double fac = std::ceil((double)index / (uniqueSpanSize_[0] + 1));
132  index -= fac * (uniqueSpanSize_[0] - 1);
133  }
134  if (surfFixedDir == 0) {
135  const int fac2 = std::floor((double)(index) / ((uniqueSpanSize_[1]) * uniqueSpanSize_[0]));
136  index -= fac2 * (uniqueSpanSize_[0] * (uniqueSpanSize_[1] - 1));
137  }
138 
139  std::array<int, dim> boundarys;
140  std::ranges::transform(uniqueSpanSize_, boundarys.begin(), [](auto& i) { return i * 2; });
141 
142  // move back to correct index, e.g. the first y - boundary index comes after all x-boundary indices
143  index += std::accumulate(boundarys.begin(), boundarys.begin() + surfFixedDir, 0);
144  } else if constexpr (dim == 3) {
145  if (surfFixedDir == 0) {
146  const int fac = (index + 1) / (uniqueSpanSize_[0] + 1);
147  index -= fac * (uniqueSpanSize_[0] - 1);
148  } else if (surfFixedDir == 1) {
149  const int fac = (index + uniqueSpanSize_[0]) / (uniqueSpanSize_[0] * (uniqueSpanSize_[1] + 1));
150  index -= fac * (uniqueSpanSize_[0] * (uniqueSpanSize_[1] - 1));
151  } else if (surfFixedDir == 2) {
152  const int fac = index / ((uniqueSpanSize_[0] * uniqueSpanSize_[1]) * (uniqueSpanSize_[2]));
153  index -= fac * (uniqueSpanSize_[1] * uniqueSpanSize_[0] * (uniqueSpanSize_[2] - 1));
154  }
155 
156  std::array<int, dim> boundarys;
157  std::ranges::fill(boundarys, 2);
158  for (int i = 0; i < dim; ++i)
159  for (int k = 0; k < dim; ++k) {
160  if (i == k) continue;
161  boundarys[i] *= uniqueSpanSize_[k];
162  }
163 
164  // move back to correct index, e.g. the first y - boundary index comes after all x-boundary indices
165  index += std::accumulate(boundarys.begin(), boundarys.begin() + surfFixedDir, 0);
166  }
167  assert(index >= 0);
168  return index;
169  }
170  __builtin_unreachable();

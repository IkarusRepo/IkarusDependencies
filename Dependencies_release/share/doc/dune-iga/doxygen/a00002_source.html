78  template <std::ranges::random_access_range Range>
79  static auto basisFunctions(ScalarType u, Range&& knots, const int p, std::optional<int> spIndex = std::nullopt) {
80  assert(std::ranges::count(knots.begin(), knots.begin() + p + 1, knots.front()) == p + 1);
81  assert(std::ranges::count(knots.end() - p - 1, knots.end(), knots.back()) == p + 1);
82  assert(spIndex < knots.size() - p - 1);
83  DynamicVectorType N;
84  N.resize(p + 1, 0.0);
85  if (Dune::FloatCmp::eq(u, knots.back())) // early exit
86  {
87  N.back() = 1;
88  return N;
89  } else if (Dune::FloatCmp::eq(u, knots.front())) {
90  N.front() = 1;
91  return N;
92  }
93 
94  const int sp = spIndex ? spIndex.value() : findSpan(p, u, knots);
95  using namespace std::ranges;
96  auto lDiff = transform_view(reverse_view(std::views::counted(knots.begin() + sp + 1 - p, p)), [&u](auto& kn) { return u - kn; });
97  auto rDiff = transform_view(std::views::counted(knots.begin() + sp + 1, p), [&u](auto& kn) { return kn - u; });
98 
99  N[0] = ScalarType(1);
100 
101  int j = 0;
102  for (auto lDp = lDiff.begin(); lDp != lDiff.end(); lDp += j + 2, ++j) {
103  ScalarType saved{};
104  for (int r = 0; auto&& rD : rDiff | std::views::take(j + 1)) {
105  const auto& lD = (*lDp);
106  const ScalarType tmp = N[r] / (rD + lD);
107  N[r++] = saved + rD * tmp;
108  saved = lD * tmp;
109  --lDp;
110  }
111  N[j + 1] = saved;
112  }
113  for ([[maybe_unused]] auto& Ni : N)
114  assert(Dune::FloatCmp::ge(Ni, 0.0)); // The basis functions are always >=0!
115 
116  return N;
117  }

128  static auto basisFunctionDerivatives(ScalarType u, const std::vector<ScalarType>& knots, const int p, const int derivativeOrder,
129  std::optional<int> spIndex = std::nullopt) {
130  assert(spIndex < knots.size() - p - 1);
131  const int order = p + 1;
132  const int sp = spIndex ? spIndex.value() : findSpan(p, u, knots);
133  using namespace std::ranges;
134  auto lDiff = transform_view(reverse_view(std::views::counted(begin(knots) + sp + 1 - p, p)), [&u](auto& kn) { return u - kn; });
135  auto rDiff = transform_view(std::views::counted(begin(knots) + sp + 1, p), [&u](auto& kn) { return kn - u; });
136 
137  DynamicMatrixType dN(derivativeOrder + 1, order);
138 
139  std::vector<ScalarType> left(order);
140  std::vector<ScalarType> right(order);
141  DynamicMatrixType ndu(order, order);
142 
143  ndu[0][0] = 1.0;
144  {
145  int j = 0;
146  for (auto lDp = lDiff.begin(); lDp != lDiff.end(); lDp += j + 2, ++j) {
147  ScalarType saved{};
148  for (int r = 0; auto&& rD : rDiff | std::views::take(j + 1)) {
149  const auto& lD = (*lDp);
150  /* Lower triangle */
151  ndu[j + 1][r] = rD + lD;
152  ScalarType tmp = ndu[r][j] / ndu[j + 1][r];
153  /* Upper triangle */
154  ndu[r++][j + 1] = saved + rD * tmp;
155  saved = lD * tmp;
156  --lDp;
157  }
158  ndu[j + 1][j + 1] = saved;
159  }
160  }
161  for (int j = p; j >= 0; --j)
162  dN[0][j] = ndu[j][p];
163 
164  // Compute the derivatives
165  RowFixedMatrix<2> a(2, p + 1);
166  auto& a1Row = a[0];
167  auto& a2Row = a[1];
168  for (int r = 0; r <= p; ++r) {
169  a[0][0] = 1.0;
170 
171  // Compute the k-th derivative
172  for (int k = 1; k <= derivativeOrder; ++k) {
173  const int rk = r - k;
174  const int pk = p - k;
175 
176  auto& nduRowpk1 = ndu[pk + 1];
177  auto& dNcur = dN[k][r];
178 
179  if (r >= k) {
180  a2Row[0] = a1Row[0] / nduRowpk1[rk];
181  dNcur = a2Row[0] * ndu[rk][pk];
182  }
183 
184  const int j1 = (rk >= -1) ? 1 : -rk;
185  const int j2 = (r - 1 <= pk) ? k - 1 : p - r;
186 
187  for (int j = j1; j <= j2; ++j) {
188  a2Row[j] = (a1Row[j] - a1Row[j - 1]) / nduRowpk1[rk + j];
189  dNcur += a2Row[j] * ndu[rk + j][pk];
190  }
191 
192  if (r <= pk) {
193  a2Row[k] = -a1Row[k - 1] / nduRowpk1[r];
194  dNcur += a2Row[k] * ndu[r][pk];
195  }
196  std::swap(a2Row, a1Row); // Switch rows
197  }
198  }
199  /* Multiply through by the correct factors */
200  /* (Eq. [2.9]) */
201  for (int r = p, k = 1; k <= derivativeOrder; ++k) {
202  for (int j = p; j >= 0; --j)
203  dN[k][j] *= r;
204  r *= p - k;
205  }
206  return dN;
207  }

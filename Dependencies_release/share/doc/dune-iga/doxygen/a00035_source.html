87  : NURBSGridView_(&gridView), directIndex_(directIndex), patchID_{patchID}, parType_{PartitionType::InteriorEntity} {
88  intersections_ = std::make_shared<std::vector<Intersection>>();
89  intersections_->reserve(this->subEntities(1));
90  for (int innerLocalIndex = 0, outerLocalIndex = 1; innerLocalIndex < this->subEntities(1); ++innerLocalIndex) {
91  const auto& eleNet = NURBSGridView_->getPatch(patchID_).elementNet_;
92  auto multiIndex = eleNet->directToMultiIndex(directIndex_);
93  multiIndex[static_cast<int>(std::floor(innerLocalIndex / 2))]
94  += ((innerLocalIndex % 2) ? 1 : Impl::noNeighbor); // increase the multiIndex depending where the outer element should lie
95  auto directOuterIndex = (eleNet->isValid(multiIndex)) ? eleNet->index(multiIndex) : Impl::noNeighbor;
96  intersections_->emplace_back(innerLocalIndex, outerLocalIndex, directIndex_, directOuterIndex, *NURBSGridView_);
97  outerLocalIndex += ((innerLocalIndex - 1) % 2) ? -1 : 3;
98  }

388  void update ()
389  {
390  n = 0;
391 
392  std::fill(offsets.begin(),offsets.end(),Index(0));
393  std::fill(blocks.begin(),blocks.end(),Index(0));
394 
395  for (unsigned int codim = 0; codim <= GV::dimension; ++codim)
396  {
397  // walk over all geometry types in the codimension
398  for (const GeometryType& gt : is.types(codim)) {
399  Index offset;
400  size_t block = layout()(gt, GV::Grid::dimension);
401 
402  // if the geometry type is contained in the layout, increment offset
403  // and store geometry type
404  if (block) {
405  offset = n;
406  n += is.size(gt) * block;
407  myTypes_[codim].push_back(gt);
408  }
409  else {
410  offset = invalidOffset;
411  }
412 
413  offsets[GlobalGeometryTypeIndex::index(gt)] = offset;
414  blocks[GlobalGeometryTypeIndex::index(gt)] = block;
415  }
416  }
417  }

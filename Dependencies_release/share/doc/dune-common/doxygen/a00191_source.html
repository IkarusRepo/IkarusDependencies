676  MessageInformation dummy;

709  sendBufferSizes_[neighbour] += tsize;

743  // save the old neighbours
744  std::size_t noOldNeighbours = remoteIndices_.neighbours();

746  sendBufferSizes_ = new std::size_t[noOldNeighbours];

844
